285 inline VectorT RandomPrototype<EngineT>::GaussianMultiVariate(const VectorT& mean, const MatrixT& cholesky)
286 {
287  LOG_DEFINE("vmcmc.random");
288  LOG_ASSERT( mean.size() == cholesky.size1() );
289 
290  VectorT noise( mean.size() );
291 
292  std::normal_distribution<typename VectorT::value_type> dist;
293  for (size_t i = 0; i < noise.size(); ++i)
294  noise[i] = dist(*this);
295 
296  return mean + ublas::prod(noise, cholesky);
